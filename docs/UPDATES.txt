====================================
ИСПОЛЬЗОВАНИЕ TOAST
====================================

import { toastStore } from './lib/toastStore.js'

// Показать уведомление
toastStore.show('Message', 'info')       // синий
toastStore.show('Success!', 'success')   // зелёный
toastStore.show('Error!', 'error')       // красный
toastStore.show('Warning!', 'warning')   // оранжевый

// С кастомной длительностью (ms)
toastStore.show('Message', 'info', 5000)
